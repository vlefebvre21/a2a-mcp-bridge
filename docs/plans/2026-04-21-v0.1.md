# v0.1 Implementation Plan — a2a-mcp-bridge

> **For Hermes:** Use `subagent-driven-development` skill to execute this plan task-by-task.

**Goal:** Ship an MCP server providing `agent_send`, `agent_inbox`, `agent_list` backed by SQLite, installable via `pipx install a2a-mcp-bridge`.

**Architecture:** Single-process MCP stdio server. A thin tool layer wraps a SQLite repository. Agent identity supplied by the client at MCP init time. No auth, no HTTP in v0.1.

**Tech Stack:** Python 3.11+, `mcp` SDK, Pydantic v2, Typer CLI, stdlib `sqlite3`, pytest + pytest-asyncio, ruff, mypy strict.

**Reference spec:** [`docs/spec/v0.1.md`](../spec/v0.1.md).

```python
from __future__ import annotations
import re
from datetime import datetime
from typing import Annotated, Any
from pydantic import BaseModel, Field, field_validator

AGENT_ID_PATTERN = re.compile(r"^[a-z0-9][a-z0-9_-]{0,63}$")
MAX_BODY_BYTES = 64 * 1024
MAX_METADATA_BYTES = 4 * 1024


class AgentId:
    @classmethod
    def validate(cls, value: str) -> str:
        if not isinstance(value, str) or not AGENT_ID_PATTERN.match(value):
            raise ValueError(f"invalid agent_id: {value!r}")
        return value


class Message(BaseModel):
    id: str
    sender_id: str
    recipient_id: str
    body: str = Field(max_length=MAX_BODY_BYTES)
    metadata: dict[str, Any] | None = None
    created_at: datetime
    read_at: datetime | None = None

    @field_validator("sender_id", "recipient_id")
    @classmethod
    def _validate_agent_id(cls, v: str) -> str:
        return AgentId.validate(v)


class AgentRecord(BaseModel):
    agent_id: str
    first_seen_at: datetime
    last_seen_at: datetime
    online: bool
    metadata: dict[str, Any] | None = None


class SendResult(BaseModel):
    message_id: str
    sent_at: datetime
    recipient: str
```

**Step 4: Implement** `src/a2a_mcp_bridge/store.py`:

```python
from __future__ import annotations
import sqlite3
from datetime import datetime, timedelta, timezone
from pathlib import Path
from .models import AgentRecord

SCHEMA_PATH = Path(__file__).parent / "schema.sql"


class Store:
    def __init__(self, db_path: str):
        self.db_path = db_path
        self._conn = sqlite3.connect(db_path, isolation_level=None)
        self._conn.row_factory = sqlite3.Row
        self._conn.execute("PRAGMA foreign_keys = ON")
        self._conn.execute("PRAGMA journal_mode = WAL")

    def init_schema(self) -> None:
        self._conn.executescript(SCHEMA_PATH.read_text())

    def upsert_agent(self, agent_id: str, metadata: dict | None = None) -> None:
        now = datetime.now(timezone.utc).isoformat()
        self._conn.execute(
            """
            INSERT INTO agents (id, first_seen_at, last_seen_at, metadata)
            VALUES (?, ?, ?, ?)
            ON CONFLICT(id) DO UPDATE SET last_seen_at = excluded.last_seen_at
            """,
            (agent_id, now, now, None),
        )

    def list_agents(self, active_within_days: int = 7) -> list[AgentRecord]:
        cutoff = (datetime.now(timezone.utc) - timedelta(days=active_within_days)).isoformat()
        rows = self._conn.execute(
            "SELECT id, first_seen_at, last_seen_at FROM agents WHERE last_seen_at >= ?",
            (cutoff,),
        ).fetchall()
        return [
            AgentRecord(
                agent_id=r["id"],
                first_seen_at=datetime.fromisoformat(r["first_seen_at"]),
                last_seen_at=datetime.fromisoformat(r["last_seen_at"]),
                online=False,  # wired up in server.py, not store
                metadata=None,
            )
            for r in rows
        ]
```

```python
# in store.py, add imports:
import uuid
from .models import SendResult, Message

# Add methods:
def send_message(
    self,
    sender: str,
    recipient: str,
    body: str,
    metadata: dict | None = None,
) -> SendResult:
    if sender == recipient:
        raise ValueError("TARGET_SELF")
    exists = self._conn.execute(
        "SELECT 1 FROM agents WHERE id = ?", (recipient,)
    ).fetchone()
    if not exists:
        raise ValueError(f"TARGET_UNKNOWN: {recipient}")

    message_id = uuid.uuid4().hex
    now = datetime.now(timezone.utc)
    self._conn.execute(
        """
        INSERT INTO messages (id, sender_id, recipient_id, body, metadata, created_at)
        VALUES (?, ?, ?, ?, ?, ?)
        """,
        (message_id, sender, recipient, body, None, now.isoformat()),
    )
    return SendResult(message_id=message_id, sent_at=now, recipient=recipient)


def read_inbox(
    self,
    agent_id: str,
    limit: int = 10,
    unread_only: bool = True,
) -> list[Message]:
    limit = max(1, min(limit, 100))
    if unread_only:
        self._conn.execute("BEGIN")
        rows = self._conn.execute(
            """
            SELECT id, sender_id, recipient_id, body, metadata, created_at, read_at
            FROM messages
            WHERE recipient_id = ? AND read_at IS NULL
            ORDER BY created_at ASC
            LIMIT ?
            """,
            (agent_id, limit),
        ).fetchall()
        now = datetime.now(timezone.utc).isoformat()
        ids = [r["id"] for r in rows]
        if ids:
            placeholders = ",".join("?" * len(ids))
            self._conn.execute(
                f"UPDATE messages SET read_at = ? WHERE id IN ({placeholders})",
                (now, *ids),
            )
        self._conn.execute("COMMIT")
    else:
        rows = self._conn.execute(
            """
            SELECT id, sender_id, recipient_id, body, metadata, created_at, read_at
            FROM messages
            WHERE recipient_id = ?
            ORDER BY created_at DESC
            LIMIT ?
            """,
            (agent_id, limit),
        ).fetchall()
    return [
        Message(
            id=r["id"],
            sender_id=r["sender_id"],
            recipient_id=r["recipient_id"],
            body=r["body"],
            metadata=None,
            created_at=datetime.fromisoformat(r["created_at"]),
            read_at=datetime.fromisoformat(r["read_at"]) if r["read_at"] else None,
        )
        for r in rows
    ]
```
